# 1. 动态类和动态属性

# 动态类

# 1.对象写法

/**
*1.v-bind和普通的class共存
*/
<div :class = "{active: isActive, 'class1'}"></div>

/**
*2.绑定的数据不必定义在模板中
*/
<div :class = "classObj"></div>
data() {
    return {
        classObj: {active : true, static : false}
    };
};

/**
 * 3. 可以直接绑定计算属性
 */
<div :class = "classObj"></div>
data() {
    return {
        isActive: false;
        flag: '1';
    },
    computed: {
        classObj() { 
            return {
                active : this.isActive;
                static : this.flag === '1'? true: false
            };
        };
    };
};
